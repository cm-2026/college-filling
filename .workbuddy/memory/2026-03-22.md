app.post('/api/recommend-from-db', async (req, res) => {
  const { score, subjectCombination, batch } = req.body;
  
  // 查询admission表
  const query = `
    SELECT school_name, major_name, min_score, min_rank, province
    FROM admission 
    WHERE min_score BETWEEN ? AND ?
      AND subject_type LIKE ?
      AND batch = ?
    ORDER BY min_score ASC
    LIMIT 20
  `;
  
  const [rows] = await pool.execute(query, [
    score - 30, score + 30,
    subjectType,
    batch
  ]);
  
  // 转换为推荐格式
  return rows.map(row => ({
    name: row.school_name,
    major: row.major_name,
    type: scoreDiff > 20 ? '冲' : (scoreDiff > -10 ? '稳' : '保'),
    score: row.min_score,

});
